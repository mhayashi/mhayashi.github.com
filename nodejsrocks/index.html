  • こうじゃなくて・・ 
result = query('select..');

ノンブロッキング

  • こう 
query('select..', function(result) {
  // 結果をつかう
  next(result);
});
// query の結果を待たずに次に進む
next(result);

  • から解放される
  • 頭の中がシンプル

さらにシンプルに - dnode

  • RMI(Remote Method Invocation)
    • サーバ 
var dnode = require('dnode');

dnode({
  decify : function (n,f) { f(n * 10) }
}).listen(6060);

さらにシンプルに - dnode

  • クライアント 
var dnode = require('dnode');

dnode.connect(6060, function (remote) {
  remote.decify(5, function (n) {
    console.log(n); // prints 50, woo!
  });
});
  • 裏ではWebsocketでやりとり
